#include <stdio.h>

int somaRec(int n, int a){
    if(n == 0){
        return a;
    }else{
        return somaRec(n-1, n+a);
    }
}

int soma(int n){
    return somaRec(n, 0);
}

int main()
{
    int numero, resul;
    
    scanf("%i", &numero);
    
    resul = soma(numero);
    
    printf("Resultado da soma: %i", resul);
    
    return 0;
}
